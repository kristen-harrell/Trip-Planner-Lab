CREATE DATABASE TripPlanner
USE TripPlanner
CREATE TABLE Trip
(Id INT NOT NULL PRIMARY KEY IDENTITY (1,1), 
[Starting Destination] NVARCHAR(25) NOT NULL,
[Ending Destination] NVARCHAR(25) NOT NULL,
[Start Date] DATE NOT NULL,
[End Date] DATE NOT NULL,
Budget FLOAT,
[Mode of Travel] NVARCHAR(15) NOT NULL)
CREATE TABLE Passport
(Id INT NOT NULL PRIMARY KEY IDENTITY (1,1),
Address NVARCHAR(50) NOT NULL,
[Expiration Date] DATE NOT NULL)
CREATE TABLE Hotel
(Id INT NOT NULL PRIMARY KEY IDENTITY (1,1), 
[Room Number] NCHAR(5) NOT NULL,
[Contact Name] NVARCHAR(25) NOT NULL,
[Contact Address] NVARCHAR (50) NOT NULL,
[Contact Phone Number] NVARCHAR(15),
[Check in date] DATE NOT NULL,
[Check out date] DATE NOT NULL,
[Cost per night] FLOAT NOT NULL)
CREATE TABLE [User]
(Id INT NOT NULL PRIMARY KEY IDENTITY (1,1),
[Name] NVARCHAR(25) NOT NULL,
[Address] NVARCHAR(50) NOT NULL,
[Phone Number] NVARCHAR(50) NOT NULL,
[DOB] DATE NOT NULL,
[HotelId] INT FOREIGN KEY REFERENCES Hotel(Id),
[TripId] INT FOREIGN KEY REFERENCES Trip(Id),
[PassportId] INT UNIQUE FOREIGN KEY REFERENCES Passport(Id))
ALTER TABLE Trip ADD UserId INT FOREIGN KEY REFERENCES [User](Id)
ALTER TABLE Hotel ADD UserId INT FOREIGN KEY REFERENCES [User](Id)
